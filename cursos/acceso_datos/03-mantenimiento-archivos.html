Código: Escritura y Lectura

# Escritura Segura con 'with'
with open('entidad.txt', 'a', encoding='utf-8') as f:
    f.write("Dato 1 | Dato 2 | Dato 3\n")

# Lectura y Procesamiento
with open('entidad.txt', 'r') as f:
    for linea in f:
        datos = linea.strip().split(" | ")
        print(f"Procesando: {datos[0]}")
